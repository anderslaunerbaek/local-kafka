```sql
CREATE STREAM CONSUMPTIONSCHEMA (
  ID VARCHAR KEY,
  CONSUMPTION_VALUE DOUBLE,
  DT_START TIMESTAMP,
  DT_END TIMESTAMP 
) WITH (
  KAFKA_TOPIC = 'CONSUMPTIONSCHEMA',
  PARTITIONS = 20, 
  VALUE_FORMAT = 'avro'
);   
CREATE STREAM PRICESCHEMA (
  ID VARCHAR KEY,
  PRICE_VALUE DOUBLE,
  DT_START TIMESTAMP,
  DT_END TIMESTAMP 
) WITH (
  KAFKA_TOPIC = 'PRICESCHEMA',
  PARTITIONS = 1, 
  VALUE_FORMAT = 'avro'
);
```
